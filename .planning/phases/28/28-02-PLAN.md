bool _hasSession = false;

@override
void initState() {
  super.initState();
  WidgetsBinding.instance.addPostFrameCallback((_) async {
    final connected = await requireSession(context, ref);
    if (!mounted) return;
    if (!connected) {
      // User dismissed gate -- go back
      context.pop();
      return;
    }
    setState(() => _hasSession = true);
  });
}
