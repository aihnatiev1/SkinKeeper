# Plan 12-01: Flutter — Local Cache Database & Cache Service

## Goal
Set up Hive local database for persistent price/inventory caching with TTL management and LRU eviction.

## Context
- Currently NO local caching — all data fetched fresh from API every app open
- SharedPreferences only used for review_service counters
- `cached_network_image` handles image caching (keep as-is)
- Need: prices (1h TTL), inventory structure (24h TTL), portfolio summary (1h TTL)
- Max cache size: 50MB with LRU eviction

### 2. Initialize Hive

File: `lib/core/cache_service.dart` (new)

```dart
import 'package:hive_flutter/hive_flutter.dart';

class CacheService {
  static late Box _priceBox;      // market_hash_name -> {prices: {...}, cachedAt: ISO}
  static late Box _inventoryBox;  // 'items' -> [{...item json...}], 'cachedAt' -> ISO
  static late Box _portfolioBox;  // 'summary' -> {...}, 'cachedAt' -> ISO
  static late Box _metaBox;       // 'lastSync' -> ISO, cache stats

  static Future<void> init() async {
    await Hive.initFlutter();
    _priceBox = await Hive.openBox('prices');
    _inventoryBox = await Hive.openBox('inventory');
    _portfolioBox = await Hive.openBox('portfolio');
    _metaBox = await Hive.openBox('cacheMeta');
  }

  // ─── Prices ───
  static Map<String, double>? getPrices(String marketHashName) {
    final entry = _priceBox.get(marketHashName);
    if (entry == null) return null;
    if (_isExpired(entry['cachedAt'], Duration(hours: 1))) return null;
    return Map<String, double>.from(entry['prices']);
  }

  static void putPrices(String marketHashName, Map<String, double> prices) {
    _priceBox.put(marketHashName, {
      'prices': prices,
      'cachedAt': DateTime.now().toIso8601String(),
    });
  }

  // ─── Inventory ───
  static List<Map<String, dynamic>>? getInventory() {
    final cachedAt = _inventoryBox.get('cachedAt');
    if (cachedAt == null) return null;
    if (_isExpired(cachedAt, Duration(hours: 24))) return null;
    final items = _inventoryBox.get('items');
    if (items == null) return null;
    return List<Map<String, dynamic>>.from(
      (items as List).map((e) => Map<String, dynamic>.from(e))
    );
  }

  static void putInventory(List<Map<String, dynamic>> items) {
    _inventoryBox.put('items', items);
    _inventoryBox.put('cachedAt', DateTime.now().toIso8601String());
  }

  // ─── Portfolio ───
  static Map<String, dynamic>? getPortfolio() {
    final entry = _portfolioBox.get('summary');
    if (entry == null) return null;
    final cachedAt = _portfolioBox.get('cachedAt');
    if (_isExpired(cachedAt, Duration(hours: 1))) return null;
    return Map<String, dynamic>.from(entry);
  }

  static void putPortfolio(Map<String, dynamic> summary) {
    _portfolioBox.put('summary', summary);
    _portfolioBox.put('cachedAt', DateTime.now().toIso8601String());
  }

  // ─── Meta ───
  static DateTime? get lastSync {
    final v = _metaBox.get('lastSync');
    return v != null ? DateTime.parse(v) : null;
  }

  static set lastSync(DateTime? dt) {
    _metaBox.put('lastSync', dt?.toIso8601String());
  }

  static String get lastSyncLabel {
    final ls = lastSync;
    if (ls == null) return 'Never synced';
    final diff = DateTime.now().difference(ls);
    if (diff.inMinutes < 1) return 'Just now';
    if (diff.inMinutes < 60) return '${diff.inMinutes}m ago';
    if (diff.inHours < 24) return '${diff.inHours}h ago';
    return '${diff.inDays}d ago';
  }

  // ─── Eviction ───
  static Future<void> evictIfNeeded({int maxBytes = 50 * 1024 * 1024}) async {
    // Hive doesn't expose box size easily; compact boxes periodically
    await _priceBox.compact();
    await _inventoryBox.compact();
    await _portfolioBox.compact();
  }

  static Future<void> clearAll() async {
    await _priceBox.clear();
    await _inventoryBox.clear();
    await _portfolioBox.clear();
    await _metaBox.clear();
  }

  // ─── Helpers ───
  static bool _isExpired(String? cachedAt, Duration ttl) {
    if (cachedAt == null) return true;
    final cached = DateTime.parse(cachedAt);
    return DateTime.now().difference(cached) > ttl;
  }
}
```

### 3. Init Cache in main.dart

File: `lib/main.dart`

Add `await CacheService.init();` before `runApp()`.

### 4. Clear Cache on Logout

File: `lib/features/auth/steam_auth_service.dart`

In `AuthNotifier.logout()`, call `CacheService.clearAll()`.
