Wire the typed AppError hierarchy to the services that still use raw duck-typed errors. export async function steamRequest<T = unknown>(opts: SteamRequestOptions): Promise<SteamResponse<T>>;

```typescript
// SteamRequestOptions:
//   url: string
//   method?: "GET" | "POST"
//   cookies?: { steamLoginSecure: string; sessionId: string; webTradeEligibility?: string }
//   params?: Record<string, string>
//   data?: Record<string, string> | string
//   headers?: Record<string, string>
//   followRedirects?: boolean   // false → maxRedirects:0
//   validateStatus?: (s: number) => boolean
//   timeout?: number
//   maxRetries?: number
//   contentType?: string

// SteamResponse<T>: { status: number; data: T; headers: Record<string, string | string[]> }

export class SteamSessionError extends Error {
  readonly code = "SESSION_EXPIRED";
  // Thrown by steamRequest() when redirected to Steam login
}
export class SteamRequestError extends Error {
  readonly httpStatus: number;
  // Thrown by steamRequest() after all retries exhausted
}
```
